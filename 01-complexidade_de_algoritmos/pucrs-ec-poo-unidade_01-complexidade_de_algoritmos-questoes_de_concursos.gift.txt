    for(int y = x; y &lt;= n; y++)

	~<code>cont \= n<sup>2</sup> + 1</code>#

	=<code>cont \= n(n + 1)/2</code>#
	~<code>cont \= n<sup>2</sup></code>#
	~<code>cont \= n + 1</code>#
}
